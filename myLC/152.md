# 152 成绩最大子数组

## 不爱睡觉的大猪

```C++
class Solution{
    public:
    int maxProduct(vector<int>& a){
        using ll = long long;

        const ll INF = 1e18;
        ll ret = -INF;
        ll val = 1LL;
        for(int l = 0, r = 0; l < a.size(); ){
            if(r == a.size() || a[r] == 0){
                // [l, r)
                while(l < r){
                    ret = max(ret, val);
                    val /= a[l++];
                }
                if(r < a.size() && a[r] == 0){
                    ret = max(ret, 0LL);
                }
                l++;
                r++;
            }else{
                val *= a[r++];
                ret = max(ret, val);
            }
        }
        return ret;
    }
};
```
