# 048 旋转图像

## 不爱睡觉的大猪

```C++
class Solution{
    public:
    void rotate(vector<vector<int>>& g){
        int n = g.size();

        // 按层处理（按圈）
        for(int L = n; L >= 0 ; L -= 2){
            int row = (n-L) >> 1;
            int col = row;

            // 当前这一层，顶行的前 L-1 个元素
            for(int k=0; k < L-1; k++){
                int r = row;
                int c = col +k;

                int tmp = g[r][c];
                // 从(r, c)开始找循环节，循环节的长度一定是4
                for(int i = 0; i < 4; i++){
                    int rr = c;
                    int cc = n - 1 - r;
                    
                    // g[r][c] -> g[rr][cc]
                    // 备份g[rr][cc]
                    swap(tmp, g[rr][cc]);
                    r = rr;
                    c = cc;
                }
            }
        }
    }
}
```
