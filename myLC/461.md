# 461 汉明距离

## 不爱睡觉的大猪

```C++
class Solution{
    public:
    int hammingDistance(int x, int y){
        // 枚举每一位
        // 32， 31
        // [0, 30]
        // bitx, bity, bitx!=bity, cnt++
        int cnt = 0;
        for(int i = 0; i < 31; i++){
            int bx = (x >> i) & 1;
            int by = (y >> i) & 1;
            if(bx != by) cnt++;
        }
        return cnt;
    }
};
```

```C++
class Solution{
    public:
    int hammingDistance(int x, int y){
        int v = x^y;
        int cnt = 0;
        while(v){
            cnt += v & 1;
            v >>= 1;
        }
        return cnt;
    }
};
```
