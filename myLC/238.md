# 238 除自身以外数组的乘积

## 不爱睡觉的大猪

```C++
class Solution{
    public:
    vector<int> productExceptSelf(vector<int>& a){
        int n = a.size();
        vector<int> b(n, 1);

        int s = 1; // 前缀/后缀乘积

        // 从上往下递推
        for(int i = 0; i < n; i++){
            b[i] *= s;
            s *= a[i];
        }

        s = 1;

        // 从下往上递推
        for(int i = n-1; ~i; i--){
            b[i] *= s;
            s *= a[i];
        }
        return b;
    }
};
```
