# 056 合并区间

## 不爱睡觉的大猪

```C++
class Solution{
    public:
    vector<vector<int>> merge(vector<vector<int>>& a){
        sort(a.begin(), a.end());

        vector<vector<int>> ret;

        int l = a[0][0], r = a[0][1];
        for(int i = 1; i < a.size(); i++){
            if(a[i][0] > r){
                ret.push_back({l, r});
                l = a[i][0], r = a[i][1];
            }else{
                r = max(r, a[i][1]);
            }
        }
        ret.push_back({l, r});
        return ret;
    }
};
```
