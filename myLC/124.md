# 124 二叉树中的最大路径和

## 不爱睡觉的大猪

```C++
class Solution{
    public:
    const int INF = 0x3f3f3f3f;
    int ret;

    int dfs(TreeNode* root){
        if(!root) return 0;
        int lret = dfs(root -> left);
        if(lret < 0) lret = 0;

        int rret = dfs(root -> right);
        if(rret < 0) rret = 0;

        ret = max(ret, root -> val + lret + rret);

        return root -> val + max(lret, rret);
    }

    int maxPathSum(TreeNode* root){
        ret = -INF;
        dfs(root);
        return ret;
    }
};
```
