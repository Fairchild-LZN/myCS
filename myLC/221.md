# 最大正方形

## 不爱睡觉的大猪

```C++
const int N = 305;
int f[N][N];

class Solution{
    public:
    int maximalSquare(vector<vector<char>>& g){
        int n = g.size(), m = g.front().size();

        memset(f, 0, sizeof f);

        int ret = 0;
        for(int i = 1; i <= n; i++){
            for(int j = 1; j <= m; j++){
                if(g[i-1][j-1] == '0') continue;
                f[i][j] = min(f[i-1][j-1], min(f[i][j-1], f[i-1][j])) + 1; // 边长
                ret = max(ret, f[i][j]);
            }
        }

        return ret * ret;
    }
};
```
