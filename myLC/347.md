# 347 前K个高频元素

## 不爱睡觉的大猪

```C++
class Solution{
    public:
    vector<int> topKFrequent(vector<int>& nums, int k){
        unordered_map<int, int> data;
        for(int x : nums) data[x] += 1;

        priority_queue<pair<int, int>> pq;
        for(auto& p : data) pq.push({p.second, p.first});

        vector<int> ret(k);
        for(int i = 0; i < k; i++){
            ret[i] = pq.top().second;
            pq.pop();
        }

        return ret;
    }
};
```
