# 053 最大子序和

## 不爱睡觉的大猪

```C++
// 前缀和为非正数，对结果没有帮助，可以舍弃
class Solution{
    public:
    int maxSubArray(vector<int>& a){
        long long sum = 0;
        int n = a.size();

        long long ret = -100000;
        for(int i = 0; i < n; i++){
            sum += a[i];
            ret = max(ret, sum);
            if(sum <= 0) sum = 0;
        }
        return ret;
    }
};
```
