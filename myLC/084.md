# 084 柱状图中最大的矩形

## 不爱睡觉的大猪

```C++
class Solution{
    public:
    int largestRectangleAre(vector<int>& h){
        int n = h.size();

        vector<int> w(n, 1);
        stack<int> stk;
        for(int i = 0; i < n; i++){
            while(!stk.empty() && h[stk.top()] >= h[i]){
                stk.pop();
            }

            if(stk.empty()) w[i] += i - 1;
            else w[i] += i - stk.top() - 1;
            stk.push(i);
        }

        while(!stk.empty()) stk.pop();

        int ret = 0;
        for(int i = n - 1; ~i; i--){
            while(!stk.empty() && h[stk.top()] >= h[i]){
                stk.pop();
            }

            if(stk.empty()) w[i] += n - 1 - i;
            else w[i] += stk.top() - i - 1;
            stk.push(i);

            ret = max(ret, h[i] * w[i])
        }
        return ret;
    }
};
```
