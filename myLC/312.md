# 312 戳气球

## 不爱睡觉的大猪

```C++
const int N = 505;
int f[N][N];

class Solution{
    public:
    int maxCoins(vector<int>& b){
        int n = b.size();
        vector<int> a(n+2);
        for(int i = 1; i <= n; i++){
            a[i] = b[i-1];
        }
        a[0] = a[n+1] = 1;

        memset(f, 0, sizeof f);

        for(int i = 1; i <= n; i++){
            f[i][i] = a[i-1] * a[i] * a[i+1];
        }

        for(int len = 2; len <= n; len++){
            for(int i = 1; i + len -1 <= n; i++){
                int j = i + len - 1;

                for(int k = i; k <= j; k++){
                    f[i][j] = max(f[i][j], f[i][k-1] + f[k+1][j] + a[i-1]*a[k]*a[j+1]);
                }
            }
        }
        return f[1][n];
    }
};
```
