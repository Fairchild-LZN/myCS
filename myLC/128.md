# 128 最长连续序列

## 不爱睡觉的大猪

```C++
class Solution{
    public:
    int longestConsecutive(vector<int>& a){
        unordered_map<int, int> data;
        for(int x : a) data[x] = 1;

        int ret = 0;
        for(auto& p : data) if(p.second){
            int val = p.first, length = 1;

            for(int i = 1; data.count(val - i) && data[val - i]; i++){
                length++, data[val - i] = 0;
            }

            for(int i = 1; data.count(val + i) && data[val + i]; i++){
                length++, data[val + i] = 0;
            }

            ret = max(ret, length);
        }
        return ret;
    }
};
```
