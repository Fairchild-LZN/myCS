# 085 最大矩形

## 不爱睡觉的大猪

```C++
class Solution{
    public:
    int maximalRectangle(vector<vector<char>>& g){
        int n = g.size();
        if(!n) return 0;
        int m = g.front().size();
        if(!m) return 0;

        int ret = 0;

        vector<int> x(n+1), y(m+1);

        // 以第 i 行为底，构造每一列的柱子高度，保存在 x[0:n-1]中
        // 可以根据第 i 行的柱子高度递推第 i+1 行的柱子高度，第 i+1 行的柱子高度保存在 y[0:m-1] 中
        // 继续往下考虑的时候，交替 x y

        for(int r = 0; r < n; r++){
            // 枚举以第 r 行作为底
            for(int c = 0; c < m; c++){
                if(g[r][c] == '0') y[c] = 0;
                else y[c] = x[c] + 1;
            }
            y[m] = -1;

            stack<int> stk;
            for(int c = 0; c < y.size(); c++){
                while(!stk.empty() && y[stk.top()] > y[c]){
                    int idx = stk.top();
                    stk.pop();

                    // 宽度：[0, c-1], c
                    if(stk.empty()) ret = max(ret, c * y[idx]);
                    // 宽度：(stk.top(), c-1], c-1-stk.top()
                    else ret = max(ret, (c - 1 - stk.top()) * y[idx]);
                }
                stk.push(c);
            }
            swap(x, y);
        }
        return ret;
    }
};
```
