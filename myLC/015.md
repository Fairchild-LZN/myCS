# 015 三数之和

## 不爱睡觉的大猪

- 先对数组升序排列
- 暴力枚举一个元素  a[i], T是目标值
- 双指针开始寻找    x+y = T-a[i]

```C++
class Solution{
    public:
    vector<vector<int>> threeSum(vector<int>& a){
        vector<vector<int>> ret;

        sort(a.begin(), a.end());

        // 暴力枚举，确定第一个元素
        // 避免暴力枚举了同一个元素

        int preid = -1;
        for(int i = 0; i < n; i++){
            // a[i] 暴力枚举的第一个元素

            // 避免暴力枚举了同一个元素
            if(preid != -1 && a[i] == a[preid]){
                continue;
            }
            
            int l = i + 1;
            int r = n - 1;
            while(l < r){
                int sum = a[i] + a[l] + a[r];
                if(sum == 0){
                    ret.push_back({a[i], a[l], a[r]});
                    int v = a[l];
                    while(l < r && a[l] == v){
                        l++;
                    }
                }else if(sum < 0){
                    l++;
                }else{
                    r--;
                }
            }

            preid = i;
        }

        return ret;
    }
}
```
