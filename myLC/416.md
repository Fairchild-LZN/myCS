# 416 分割等和子集

## 不爱睡觉的大猪

```C++
class Solution{
    public:
    // f[i][s] 是否能从前i个元素中选择一些元素，使得他们的和为s
    // true 可以选出这些元素
    // false 不能

    // 答案f[n-1][t], t = sum/2, sum整个数组的总和
    // f[i][s] = -1; 表示这个状态还没有被计算过
    // f[i][s] = 1; 等价于true
    // f[i][s] = 0; 等价于false
    vector<vector<int>> f;
    vector<int> a;
    int n;

    int dfs(int m, int t){
        if(t == 0) return 1;
        if(m == -1 || t < 0) return 0;
        if(f[m][t] != -1) return f[m][t];

        f[m][t] = 0;
        // 选、不选
        f[m][t] = dfs(m-1, t - a[m]) || dfs(m-1, t);

        return f[m][t];
    }

    bool canPartition(vector<int>& a){
        n = a.size();
        this -> a = a;
        int sum = 0;    // 计算数组总和

        for(int x:a) sum += x;

        if(sum & 1) return false; // 数组总和为奇数

        int t = sum / 2;

        // 初始化状态数组
        f.assign(n, vector<int>(t+1, -1));

        return dfs(n-1, t);
    }
};
```
