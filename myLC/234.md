# 234 回文链表

## 不爱睡觉的大猪

```C++
class Solution{
    public:
    bool isPalindrome(ListNode* head){
        ListNode* ptr = head;
        int n = 0;
        while(ptr) n++, ptr = ptr->next;

        int half = n/2;
        ListNode* pre = nullptr;
        ptr = head;

        for(int i = 0; i < half; i++){
            ListNode* node = ptr -> next;
            ptr -> next = pre;
            pre = ptr;
            ptr = node;
        }

        if(n & 1) ptr = ptr -> next;

        while(ptr){
            if(ptr->val != pre->val) return false;
            ptr = ptr -> next;
            pre = pre -> next;
        }

        return true;
    }
};
```
