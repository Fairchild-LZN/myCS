# 032 最长有效括号

## 不爱睡觉的大猪

```C++
class Solution{
    public:
    int longestValidParentheses(string s){
        int n = s.length();
        stack<int> stk;

        stk.push(-1);
        int ret = 0;
        for(int i = 0; i < n; i++){
            if(s[i] == '(') stk.push(i);
            else{
                if(stk.top() == -1 || s[stk.top()] == ')'){
                    stk.push(i);
                }else{
                    stk.pop(); // [l,i]
                    int b = stk.top();
                    ret = max(ret, i - b);
                }
            }
        }
        return ret;
    }
};
```
